       ```javascript
       function readJsonBody(req) {
         return new Promise((resolve, reject) => {
           let body = '';
           req.on('data', chunk => { body += chunk; });
           req.on('end', () => {
             try { resolve(JSON.parse(body)); }
             catch (e) { reject(new Error('Invalid JSON')); }
           });
           req.on('error', reject);
         });
       }
       ```

       ```javascript
       }
       ```

       ```javascript
       async function handleApiRequest(req, res, urlPath) {
         const { method } = req;
         res.setHeader('Content-Type', 'application/json');
         res.setHeader('Access-Control-Allow-Origin', '*');

         // Health check
         if (urlPath === '/api/health' && method === 'GET') {
           res.writeHead(200);
           res.end(JSON.stringify({ status: 'ok' }));
           return;
         }

         // Templates
         if (urlPath === '/api/templates' && method === 'GET') {
           const dir = path.join(__dirname, 'data', 'templates');
           await mkdir(dir, { recursive: true });
           const files = await readdir(dir);
           const templates = await Promise.all(
             files.filter(f => f.endsWith('.json')).map(async f => {
               const content = await readFileAsync(path.join(dir, f), 'utf8');
               return JSON.parse(content);
             })
           );
           res.writeHead(200);
           res.end(JSON.stringify(templates));
           return;
         }

         if (urlPath === '/api/templates' && method === 'POST') {
           const body = await readJsonBody(req);
           const id = crypto.randomUUID();
           const filePath = path.join(__dirname, 'data', 'templates', `${id}.json`);
           await writeJsonFile(filePath, { id, ...body });
           res.writeHead(201);
           res.end(JSON.stringify({ id }));
           return;
         }

         // Reports
         if (urlPath === '/api/reports' && method === 'GET') {
           const dir = path.join(__dirname, 'data', 'reports');
           await mkdir(dir, { recursive: true });
           const files = await readdir(dir);
           const reports = await Promise.all(
             files.filter(f => f.endsWith('.json')).map(async f => {
               const content = await readFileAsync(path.join(dir, f), 'utf8');
               return JSON.parse(content);
             })
           );
           res.writeHead(200);
           res.end(JSON.stringify(reports));
           return;
         }

         if (urlPath === '/api/reports' && method === 'POST') {
           const body = await readJsonBody(req);
           const id = crypto.randomUUID();
           const filePath = path.join(__dirname, 'data', 'reports', `${id}.json`);
           await writeJsonFile(filePath, { id, ...body });
           res.writeHead(201);
           res.end(JSON.stringify({ id }));
           return;
         }

         // Unmatched API route
         res.writeHead(404);
         res.end(JSON.stringify({ error: 'Not found' }));
       }
       ```

       ```javascript
       const server = http.createServer(async (req, res) => {

       ```
